$("<button>Show password</button>").click( function() {
  var pInput =  $("p:eq(1) input");
  var type = pInput.attr("type");
  var nouveauType = (type == "password")? "text" : "password";
  var oldValue = pInput.val();
  pInput.replaceWith("<input type='" + nouveauType +
    "' value='" + oldValue +"' />");
})
.appendTo("p:eq(1)");
